  <script>
    const slider = document.getElementById('slider');
    const dayValue = document.getElementById('dayValue');
    const pechayModel = document.getElementById('pechayModel');

    slider.addEventListener('input', () => {
      const day = parseInt(slider.value);
      dayValue.textContent = day;

      // Simple AR Growth Logic (adjust as needed)
      const growthScale = 0.05 + (day / 365) * 0.25; // from small seedling to mature
      const greenShade = Math.min(255, 100 + day / 2);

      // Apply transformations
      pechayModel.setAttribute('scale', `${growthScale} ${growthScale} ${growthScale}`);
      pechayModel.setAttribute('material', `color: rgb(0, ${greenShade}, 0)`);

      console.log(`Day ${day} â†’ Growth Scale: ${growthScale.toFixed(3)} | Green: ${greenShade}`);
    });
  </script>
